/* Demonstra constant folding, código morto e preservação de ciclo */
/* num mesmo programa: fatorial iterativo com variável morta. */
vazio principal(vazio);

vazio principal(vazio) {
    inteiro i;
    inteiro resultado;
    inteiro morta;
    resultado = 1;
    morta = 999;
    para (i = 1; i < 6; i = i + 1) {
        resultado = resultado * i;
    }
    escrever(resultado);
}
