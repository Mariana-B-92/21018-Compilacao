/* Função recursiva: soma os dígitos de um número */
inteiro somaDigitos(inteiro n) {
    se (n <= 0) {
        retornar 0;
    } senao {
        retornar n % 10 + somaDigitos(n / 10);
    }
}

/* Função recursiva: calcula o fatorial */
inteiro fatorial(inteiro n) {
    se (n <= 1) {
        retornar 1;
    } senao {
        retornar n * fatorial(n - 1);
    }
}

/* Função que imprime um vetor de inteiros */
vazio imprimirVetor(inteiro v[], inteiro tamanho) {
    inteiro i;
    para (i = 0; i < tamanho; i = i + 1) {
        escrever(v[i]);
    }
}
