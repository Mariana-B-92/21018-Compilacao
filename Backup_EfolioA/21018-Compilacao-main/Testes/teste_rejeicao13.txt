/* Parâmetros sem Identificador em Definição de Função */
/* Regra: Na definição de uma função, todos os parâmetros têm de ter identificador. */
/* Num protótipo, o identificador é opcional. - inválido */

/* Protótipo válido: identificadores omitidos são permitidos aqui */
inteiro soma(inteiro, inteiro);
vazio principal(vazio);

/* Definição inválida: parâmetros sem identificador não são permitidos na definição */
inteiro soma(inteiro, inteiro b) {
    retornar b;
}

vazio principal(vazio) {
    escrever(soma(1, 2));
}
